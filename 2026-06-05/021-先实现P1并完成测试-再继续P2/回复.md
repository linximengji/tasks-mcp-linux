// 1. 队列数据结构（第 368-376 行）
const sessionQueues = new Map<string, Promise<void>>()

async function enqueueSession<T>(sessionKey: string, fn: () => Promise<T>): Promise<T> {
  const prev = (sessionQueues.get(sessionKey) || Promise.resolve()).catch(() => {})
  const next = prev.then(() => fn())
  sessionQueues.set(sessionKey, next.then(() => {}) as Promise<void>)
  return next
}

// 2. callClaude / callClaudeStreaming 变为队列封装（第 513, 706 行）
export async function callClaude(...) {
  return enqueueSession(sessionKey, () => _execClaude(options, retryCount))
}

// 3. 原 body 移到 _execClaude / _execClaudeStreaming
// 4. 内部递归重试改为调用 _exec* 而非 callClaude，避免死锁

### 机制

| 场景 | 行为 |
|------|------|
| 同一 session 并发消息（Msg1 + Msg2） | Msg2 等 Msg1 完成后才启动，读到更新后的 session_id |
| 不同 session（群 A + 群 B） | 不互相阻塞，并行执行 |
| session 失效重试 | 重试在队列 slot 内完成，不释放锁 |
| 前一个任务抛异常 | `.catch(() => {})` 吞噬，下一个任务正常执行 |
